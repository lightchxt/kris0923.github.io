# 基本数据类型
## int 转string
```
var i = 10
s := strconv.Itoa(i)
fmt.Printf("%v", s)
```
## string 转int
```
var s = "10"
i, err := strconv.Atoi(s)
if err != nil {
	fmt.Println(err)
	return
}
fmt.Printf("%v", i)
// int64
ii, err := strconv.ParseInt(s, 10, 32)
if err != nil {
	fmt.Println(err)
	return
}
fmt.Printf("%v", ii)
// float64
f, err := strconv.ParseInt(s, 10)
if err != nil {
	fmt.Println(err)
	return
}
fmt.Printf("%v", f)
```
